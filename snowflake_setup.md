# Introduction and Environment Setup

## Snowflake user creation
Copy these SQL statements into a Snowflake Worksheet, select all and execute them (i.e. pressing the play button).

```sql
-- Use an admin role
USE ROLE ACCOUNTADMIN;

-- Create the `transform` role
CREATE ROLE IF NOT EXISTS transform2;
GRANT ROLE TRANSFORM2 TO ROLE ACCOUNTADMIN;

-- Create the default warehouse if necessary
CREATE WAREHOUSE IF NOT EXISTS COMPUTE_WH;
GRANT OPERATE ON WAREHOUSE COMPUTE_WH TO ROLE TRANSFORM2;

-- Create our database and schemas
CREATE DATABASE IF NOT EXISTS PAYROLL_WAREHOUSE2;
CREATE SCHEMA IF NOT EXISTS PAYROLL_WAREHOUSE2.LANDING;
CREATE SCHEMA IF NOT EXISTS PAYROLL_WAREHOUSE2.MARTS;


-- Create the `dbt` user and assign to role
CREATE USER IF NOT EXISTS dbt_payroll2
  PASSWORD=''
  LOGIN_NAME=''
  MUST_CHANGE_PASSWORD=FALSE
  DEFAULT_WAREHOUSE='COMPUTE_WH'
  DEFAULT_ROLE='transform2'
  DEFAULT_NAMESPACE='PAYROLL_WAREHOUSE2.LANDING'
  COMMENT='DBT user used for data transformation';
GRANT ROLE transform2 to USER dbt_payroll2;

-- Set up permissions to role `transform`
GRANT ALL ON WAREHOUSE COMPUTE_WH TO ROLE transform2; 
GRANT ALL ON DATABASE PAYROLL_WAREHOUSE2 to ROLE transform2;
GRANT ALL ON ALL SCHEMAS IN DATABASE PAYROLL_WAREHOUSE2 to ROLE transform2;
GRANT ALL ON FUTURE SCHEMAS IN DATABASE PAYROLL_WAREHOUSE2 to ROLE transform2;
GRANT ALL ON ALL TABLES IN SCHEMA PAYROLL_WAREHOUSE2.LANDING to ROLE transform2;
GRANT ALL ON FUTURE TABLES IN SCHEMA PAYROLL_WAREHOUSE2.LANDING to ROLE transform2;


-- Setup permissions for pbi user
CREATE ROLE IF NOT EXISTS REPORTER2;
CREATE USER IF NOT EXISTS PBI2
    PASSWORD=''
    LOGIN_NAME=''
    MUST_CHANGE_PASSWORD=FALSE
    DEFAULT_WAREHOUSE='COMPUTE_WH'
    DEFAULT_ROLE='REPORTER2'
    DEFAULT_NAMESPACE='PAYROLL_WAREHOUSE2.MARTS'
    COMMENT='PowerBI user for creating reports';
GRANT ROLE REPORTER2 TO USER PBI2;
GRANT ROLE REPORTER2 TO ROLE ACCOUNTADMIN;
GRANT ALL ON WAREHOUSE COMPUTE_WH TO ROLE REPORTER2;
GRANT USAGE ON DATABASE PAYROLL_WAREHOUSE2 TO ROLE REPORTER2;
GRANT USAGE ON SCHEMA PAYROLL_WAREHOUSE2.MARTS TO ROLE REPORTER2;
GRANT SELECT ON ALL TABLES IN SCHEMA PAYROLL_WAREHOUSE2.MARTS TO ROLE REPORTER2;
GRANT SELECT ON FUTURE TABLES IN SCHEMA PAYROLL_WAREHOUSE2.MARTS TO ROLE REPORTER2;
